 $mdr = new MDR_api();

 $mdr->addParam( "command", "domain_list" );
 $mdr->addParam( "sort",    "registrant"  );
 $mdr->addParam( "order",   "1"           );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 }
 else {
  print "OK<BR>";
  print "<TABLE>";
   for($i=0;$i<=$mdr->Values["domeincount"];$i++) {
    print "<TR>";
    print "<TD>" . $mdr->Values["domein[$i]"]       . "</TD>";
    print "<TD>" . $mdr->Values["registrant[$i]"]   . "</TD>";
    print "<TD>" . $mdr->Values["admin[$i]"]        . "</TD>";
    print "<TD>" . $mdr->Values["tech[$i]"]         . "</TD>";
    print "<TD>" . $mdr->Values["verloopdatum[$i]"] . "</TD>";
    print "<TD>" . $mdr->Values["status[$i]"]       . "</TD>";
    print "</TR>";
   }
  print "</TABLE>";
 }
