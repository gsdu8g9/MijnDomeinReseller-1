Voorbeeld 1

 $mdr = new MDR_api();

 $mdr->addParam( "command", "contact_list" );
 $mdr->addParam( "sort",    "bedrijfsnaam" );
 $mdr->addParam( "order",   "1"            );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 } else {
  print "OK<BR>";
  print "<TABLE>";
  for($i=0;$i<$mdr->Values["contactcount"];$i++) {
    print "<TR>";
    print "<TD>" . $mdr->Values[ "contact_id[$i]" ]            . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_bedrijfsnaam[$i]" ]  . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_voorletter[$i]" ]    . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_tussenvoegsel[$i]" ] . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_achternaam[$i]" ]    . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_straat[$i]" ]        . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_huisnr[$i]" ]        . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_huisnrtoev[$i]" ]    . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_postcode[$i]" ]      . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_plaats[$i]" ]        . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_land[$i]" ]          . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_email[$i]" ]         . "</TD>";
    print "<TD>" . $mdr->Values[ "contact_tel[$i]" ]           . "</TD>";
    print "</TR>";
  }
  print "</TABLE>";
 }
