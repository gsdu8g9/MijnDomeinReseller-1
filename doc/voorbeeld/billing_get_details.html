Voorbeeld: billing_get_details

Voorbeeld 1

 $mdr = new MDR_api();

 $mdr->addParam( "command", "billing_get_details" );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 } else {
  print "OK<BR>";
  print "<TABLE>";
  for($i=0;$i<$mdr->Values["billing_details_count"];$i++) {
    print "<TR>";
    print "<TD>" . $mdr->Values[ "datum[$i]" ]        . "</TD>";
    print "<TD>" . $mdr->Values[ "prijs[$i]" ]        . "</TD>";
    print "<TD>" . $mdr->Values[ "domein[$i]" ]       . "</TD>";
    print "<TD>" . $mdr->Values[ "tld[$i]" ]          . "</TD>";
    print "<TD>" . $mdr->Values[ "omschrijving[$i]" ] . "</TD>";
    print "</TR>";
  }
  print "</TABLE>";
 }
