 $mdr = new MDR_api();

 $mdr->addParam( "command", "nameserver_glue_get_details" );
 $mdr->addParam( "domein",  "mijndomeinreseller"          );
 $mdr->addParam( "tld",     "nl"                          );
 $mdr->addParam( "host",    "ns1"                         );


 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 } else {
  print "OK<BR>";
  print "Actief: " . $mdr->Values["active"] . " <BR>";
  print "<TABLE>";
  for($i=1;$i<$mdr->Values["ipcount"];$i++) {
    print "<TR>";
    print "<TD>IP $i : " . $mdr->Values[ "ip[$i]" ]       . "</TD>";
    print "</TR>";
  }
  print "</TABLE>";
 }
