 $mdr = new MDR_api();

 $mdr->addParam( "command", "dns_get_details"     );
 $mdr->addParam( "domein",  "mijndomeinreseller"  );
 $mdr->addParam( "tld",     "nl"                  );


 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 } else {
  print "OK<BR>";
  print "<TABLE>";
  for($i=0;$i<$mdr->Values["recordcount"];$i++) {
    print "<TR>";
    print "<TD>" . $mdr->Values[ "record_id[$i]" ] . "</TD>";
    print "<TD>" . $mdr->Values[ "type[$i]" ]      . "</TD>";
    print "<TD>" . $mdr->Values[ "host[$i]" ]      . "</TD>";
    print "<TD>" . $mdr->Values[ "address[$i]" ]   . "</TD>";
    print "<TD>" . $mdr->Values[ "priority[$i]" ]  . "</TD>";
    print "<TD>" . $mdr->Values[ "weight[$i]" ]    . "</TD>";
    print "<TD>" . $mdr->Values[ "port[$i]" ]      . "</TD>";
    print "</TR>";
  }
  print "</TABLE>";
 }
