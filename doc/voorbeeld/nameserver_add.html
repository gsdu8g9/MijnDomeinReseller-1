 $mdr = new MDR_api();

 $mdr->addParam( "command", "nameserver_add"       );

 $mdr->addParam( "ns1",     "ns1.mijndnsserver.nl" );
 $mdr->addParam( "ns1_ip",  "82.192.92.144"         );
 $mdr->addParam( "ns2",     "ns2.mijndnsserver.nl" );
 $mdr->addParam( "ns2_ip",  "109.72.95.76"         );
 $mdr->addParam( "ns3",     "ns3.mijndnsserver.eu" );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 }
 else {
  print "OK<BR>";
  print "ns_id: " . $mdr->Values[ "ns_id" ];
 }
