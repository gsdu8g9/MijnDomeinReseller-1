Voorbeeld 1

 $mdr = new MDR_api();

 $mdr->addParam( "command", "transfer_list" );

 $mdr->DoTransaction();

 if( $mdr->Values[ "errcount" ] > 0 ) {
   print "Fout opgetreden:<BR>";

   for( $i=1;$i<=$mdr->Values[ "errcount" ];$i++ ) {
     print $mdr->Values[ "errnotxt".$i ] . " (code: " . $mdr->Values[ "errno".$i ] . ")<BR>";
   }

 }
 else {
   print "OK<BR>";
   print "<TABLE>";
   for($i=0;$i<$mdr->Values["transfercount"];$i++) {
     print "<TR>";
     print "<TD>" . $mdr->Values[ "transfer_id[$i]" ]    . "</TD>";
     print "<TD>" . $mdr->Values[ "domein[$i]" ]         . "</TD>";
     print "<TD>" . $mdr->Values[ "status[$i]" ]         . "</TD>";
     print "<TD>" . $mdr->Values[ "status_melding[$i]" ] . "</TD>";
     print "<TD>" . $mdr->Values[ "datum_invoer[$i]" ]   . "</TD>";
     print "<TD>" . $mdr->Values[ "datum_update[$i]" ]   . "</TD>";
     print "</TR>";
   }
   print "</TABLE>";
 }
